+from __future__ import annotations
+
+import argparse
+from email import policy
+from email.message import EmailMessage
+from email.parser import BytesParser
+from pathlib import Path
+from typing import Iterable
+
+
+def _is_japanese_char(ch: str) -> bool:
+    codepoint = ord(ch)
+    return (
+        0x3000 <= codepoint <= 0x303F  # punctuation
+        or 0x3040 <= codepoint <= 0x30FF  # hiragana/katakana
+        or 0x4E00 <= codepoint <= 0x9FFF  # CJK
+    )
+
+
+def restore_utf8_with_stripped_high_bits(data: bytes) -> bytes:
+    """Restore UTF-8 text where bytes >=0x80 lost the high bit."""
+    restored: list[int] = []
+    i = 0
+    length = len(data)
+    while i < length:
+        current = data[i]
+        next_one = data[i + 1] if i + 1 < length else None
+        next_two = data[i + 2] if i + 2 < length else None
+        if (
+            0x60 <= current <= 0x6F
+            and i + 2 < length
+            and next_one is not None
+            and next_two is not None
+            and next_one < 0x40
+            and next_two < 0x40
+            and next_one not in (0x0A, 0x0D)
+            and next_two not in (0x0A, 0x0D)
+        ):
+            candidate = bytes(
+                (current | 0x80, data[i + 1] | 0x80, data[i + 2] | 0x80)
+            )
+            try:
+                decoded = candidate.decode("utf-8")
+            except UnicodeDecodeError:
+                decoded = ""
+            if decoded and _is_japanese_char(decoded):
+                restored.extend(candidate)
+                i += 3
+                continue
+
+        restored.append(current)
+        i += 1
+
+    return bytes(restored)
+
+
+def _restore_plain(part: EmailMessage) -> str:
+    payload = part.get_payload(decode=True) or b""
+    fixed = restore_utf8_with_stripped_high_bits(payload)
+    text = fixed.decode(part.get_content_charset() or "utf-8")
+    part.set_content(text, subtype=part.get_content_subtype(), charset="utf-8", cte="quoted-printable")
+    return text
+
+
+def _escape_html(text: str) -> str:
+    return (
+        text.replace("&", "&amp;")
+        .replace("<", "&lt;")
+        .replace(">", "&gt;")
+    )
+
+
+def _build_html(text: str) -> str:
+    escaped = _escape_html(text)
+    return (
+        "<!DOCTYPE html>\n"
+        "<html>\n"
+        "  <head>\n"
+        '    <meta http-equiv="content-type" content="text/html; charset=UTF-8">\n'
+        "  </head>\n"
+        '  <body text="#404040" bgcolor="#FFFFFF">\n'
+        "    <pre style=\"font-family: 'Noto Sans JP', sans-serif; white-space: pre-wrap;\">"
+        f"{escaped}</pre>\n"
+        "  </body>\n"
+        "</html>\n"
+    )
+
+
+def restore_eml(path: Path) -> None:
+    message = BytesParser(policy=policy.default).parsebytes(path.read_bytes())
+    restored_plain: str | None = None
+    html_parts: list[EmailMessage] = []
+
+    for part in message.walk():
+        if part.get_content_maintype() != "text":
+            continue
+        if part.get_content_subtype() == "plain":
+            restored_plain = _restore_plain(part)
+        elif part.get_content_subtype() == "html":
+            html_parts.append(part)
+
+    if restored_plain is None:
+        restored_plain = ""
+
+    for html_part in html_parts:
+        html_part.set_content(
+            _build_html(restored_plain),
+            subtype="html",
+            charset="utf-8",
+            cte="quoted-printable",
+        )
+
+    path.write_bytes(message.as_bytes(policy=policy.SMTP))
+
+
+def parse_args(args: Iterable[str] | None = None) -> argparse.Namespace:
+    parser = argparse.ArgumentParser(description="Restore garbled forwarded EML files.")
+    parser.add_argument("eml_path", type=Path, help="Path to the .eml file to restore")
+    return parser.parse_args(args)
+
+
+def main() -> None:
+    args = parse_args()
+    restore_eml(args.eml_path)
+
+
+if __name__ == "__main__":
+    main()
